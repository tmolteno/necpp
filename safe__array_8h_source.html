102  void resize(int64_t new_length) {
103 #ifdef NEC_ERROR_CHECK
104  if (! own_data_)
105  throw new nec_exception("attempt to resize data we do not own");
106 #endif
107  if (new_length > data_size_) {
108  // We allocate resize_chunk_ more bytes than we need to avoid
109  // resizing too often.
110  data_size_ = new_length + resize_chunk_;
111  try {
112  T* new_data_ = new T[data_size_];
113  if (0 != len_)
114  std::memcpy(new_data_, data_, len_ * sizeof(T));
115 
116  delete[] data_;
117  data_ = new_data_;
118  } catch (std::bad_alloc& ba) {
119  throw new nec_exception("Error: Out of Memory ");
120  }
121  }
122  len_ = new_length;
123  }

251  safe_array(const safe_array<T>& in_array, int64_t start_index, int64_t end_index, bool in_copy_data)
252  {
253  resize_chunk_ = in_array.resize_chunk_;
254  len_ = (end_index - start_index)+1; // include the end_index element
255  rows_ = 0;
256  cols_ = 0;
257 
258  if (in_copy_data) {
259  data_ = new T[len_];
260  data_size_ = len_;
261 
262  for (int64_t i=0; i<len_; i++)
263  data_[check(i)] = in_array[start_index + i];
264 
265  own_data_ = true;
266  } else {
267  data_ = in_array.data() + start_index;
268  data_size_ = 0;
269  own_data_ = false;
270  }
271  }

273  inline int64_t check(int64_t i) const {
274 #ifdef NEC_ERROR_CHECK
275  if (i < 0 || i >= len_)
276  throw new BoundsViol("safe_array: ", i, len_);
277 #endif
278  return i;
279  }

281  inline int64_t check(int32_t row, int32_t col) const {
282 #ifdef NEC_ERROR_CHECK
283  if (row < 0 || row >= rows_)
284  throw new BoundsViol("safe_array: ", row, rows_);
285  if (col < 0 || col >= cols_)
286  throw new BoundsViol("safe_array: ", col, cols_);
287 #endif
288  return check(int64_t(col)*rows_ + row);
289  }
