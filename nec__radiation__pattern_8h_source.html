269  nec_float mean(real_array& pattern) const
270  {
271  nec_float sum = 0.0;
272  long len = pattern.size();
273  for (long i=0;i<len;i++)
274  {
275  sum += pattern[i] * _averaging_scales[i];
276  }
277  return sum/(len*2.0 / pi());
278  }
279 
280  nec_float sd(real_array& pattern, nec_float _mean) const
281  {
282  nec_float sum = 0.0;
283  long len = pattern.size();
284  for (long i=0;i<len;i++)
285  {
286  nec_float diff = pattern[i] - _mean;
287  sum += diff*diff * _averaging_scales[i];
288  }
289  return std::sqrt(sum/(len*2.0 / pi()));
290  }

425  nec_float get_power_gain_rhcp(int theta_index, int phi_index) const
426  {
427  nec_float a = get_pol_axial_ratio(theta_index, phi_index);
428  nec_float dbi = get_power_gain_tot(theta_index, phi_index);
429  if (get_pol_sense(theta_index, phi_index) == POL_RIGHT)
430  a = -a;
431 
432  nec_float f = (1-2*a+a*a)/(2*(1+a*a));
433  return dbi + 10*log10(f);
434  }
435 
438  nec_float get_power_gain_lhcp(int theta_index, int phi_index) const
439  {
440  nec_float a = get_pol_axial_ratio(theta_index, phi_index);
441  nec_float dbi = get_power_gain_tot(theta_index, phi_index);
442  if (get_pol_sense(theta_index, phi_index) == POL_RIGHT)
443  a = -a;
444 
445  nec_float f = (1+2*a+a*a)/(2*(1+a*a));
446  return dbi + 10*log10(f);
447  }
